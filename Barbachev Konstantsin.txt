1) ls  ///текущая директория///
2) cd "название папки/"  ///удостоверяемся, что мы в нужной папке///
3) Заходим в свой репозиторий на GitHub и копируем URL этого репозитория
4) git clone "URL репозитория"  /// создаем локальную копия репозитория GitHub///
5) cd "название репозитория"  ///удостоверяемся, что мы в нужной ветке (main)///
6) ls  ///список файлов///
7) git status  ///находимся в главной ветке, коммитить нечего///
8) создаем файл "имя файла"
9) git status  /// новый файл подсвечен красным ///
10) git add "имя файла" или "."  ///файл добавляется в индекс///
11) git status  ///проверяем статус нашего файла: цвет изменился на зеленый///
12) git commit -m "комментарий"  ///фиксируем изменения///
13) git push  ///отправляем на сервер///
14) удостоверяемся, что в репозитории на GitHub появился наш файл
15) git log  ///проверяем все записанные commit, в том числе и наш///
16) shift+q  ///выход в коммандную строку GitBash///
